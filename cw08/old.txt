#include <stdio.h>
#include <time.h>
#include <pthread.h>
#include <unistd.h>

pthread_t watek01,watek02;
void* fun_watka(void* cos){
    while(1){
        printf("%s ",(char*)cos);
        sleep(1);
    }
}
//end funckja watka
int main(void){
    int i;
    pthread_create(&watek01,NULL,&fun_watka,"A");
    pthread_create(&watek02,NULL,&fun_watka,"B");
    for (i=1;i<10000;i++){
        printf("%d ",i);
        fflush(stdout);
        sleep(3);
    }
}
